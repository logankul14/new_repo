#!/bin/bash

read -p "enter first number" a
read -p "enter second number" b

echo "enter your choice"

echo "1.addition"
echo "2.substraction"
echo "3.multiplication"
echo "4.division"

read choice 

case $choice in 
	1)result=$((a + b)) ;;
	2)result=$((a - b)) ;;
	3)result=$((a * b)) ;;
	4)result=$((a / b)) ;;
esac

echo "Result: $result"
